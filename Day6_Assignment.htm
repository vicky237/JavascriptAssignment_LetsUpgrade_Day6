Question 1

Write a program that will iterate over an array of colors and change the background of the page after 5 seconds 


                let colorArray = ['Black', 'Green', 'Blue', 'Yellow'];
                let COunt = 0;
                let changeBg = (() => {
        
                    if (colorArray.length >= COunt) {
                        document.body.style.backgroundColor = colorArray[COunt];
                        document.body.style.color = 'white';
                        document.body.style.fontFamily = 'Helvetica';
                        COunt += 1;
                    }
                });
                setInterval(changeBg, 5000);
